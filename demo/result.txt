变量
{
	玩家:
		0: damage_id

}

规则("chuanhuo")
{
	事件
	{
		玩家受到伤害;
		所有队伍;
		全部英雄;
	}
	条件
	{
		非(具有状态(事件玩家, 点燃)) == 真;
		具有状态(攻击方, 点燃) == 真;
	}
	动作
	{
		设置状态(事件玩家, 无, 点燃, 9999.000);
		开始持续伤害(事件玩家, 无, 9999.000, 50.000);
		设置玩家变量(事件玩家, damage_id, 上一个持续伤害效果ID);
		停止持续伤害(玩家变量(攻击方, damage_id));
		清除状态(攻击方, 点燃);
	}
}

规则("jiafen")
{
	事件
	{
		玩家阵亡;
		所有队伍;
		全部英雄;
	}
	条件
	{
		存活玩家数量(所有队伍) == 1.000;
		游戏正在进行中 == 真;
	}
	动作
	{
		修改玩家分数(所有存活玩家(所有队伍), 1.000);
		重生(所有玩家(所有队伍));
	}
}
